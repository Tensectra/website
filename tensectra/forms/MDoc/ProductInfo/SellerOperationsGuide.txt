PART 2 — THE WEBHOOK API: WHAT IT IS AND WHERE IT LIVES
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  THE QUESTION: Where does the webhook code live?
  ─────────────────────────────────────────────────

  NOT in tensectrakit-core     — that is the product you sell. Do not put ops code there.
  NOT on GitHub Actions alone  — GitHub Actions cannot receive Gumroad webhooks.
  NOT on the buyer's machine   — buyers never see seller infrastructure.

  ANSWER: A SEPARATE, LIGHTWEIGHT .NET 8 MINIMAL API — your "License Server"
  ─────────────────────────────────────────────────────────────────────────────

  You create and deploy a second repository:

    tensectra-pro/tensectrakit-license-server   (PRIVATE repo — never shown to buyers)

  This is a tiny .NET 8 Minimal API with ONE job:
    - Receive a POST from Gumroad when a sale happens
    - Parse the buyer's GitHub username from the sale data 
    - Call the GitHub API to invite that username to the correct private repo
    - Record the purchase in a SQLite database (for expiry tracking)

  The License Server is deployed to Railway (free, public HTTPS URL in 5 minutes).
  You give Gumroad that HTTPS URL as the "Ping URL" → Gumroad calls it on every sale.

PART 3 — LICENSE SERVER SOURCE CODE  (.NET 8 Minimal API — complete, production-ready)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  FILE: Program.cs  (replace entire file)

      using Microsoft.Data.Sqlite;
      using Dapper;
      using System.Text.Json;
      using System.Text.Json.Serialization;

      var builder = WebApplication.CreateBuilder(args);
      builder.Services.AddHttpClient();
      var app = builder.Build();

      // ── Database init ──────────────────────────────────────────────────────────
      var dbPath = builder.Configuration["DbPath"] ?? "licenses.db";
      using (var db = new SqliteConnection($"Data Source={dbPath}"))
      {
          db.Execute(@"
              CREATE TABLE IF NOT EXISTS Purchases (
                  Id              INTEGER PRIMARY KEY AUTOINCREMENT,
                  OrderId         TEXT    NOT NULL,
                  BuyerEmail      TEXT    NOT NULL,
                  GitHubUsername  TEXT    NOT NULL,
                  ProductName     TEXT    NOT NULL,
                  LicenseTier     TEXT    NOT NULL,
                  PurchasedAt     TEXT    NOT NULL,
                  ExpiresAt       TEXT,
                  AccessRevoked   INTEGER NOT NULL DEFAULT 0,
                  LicenseKey      TEXT    NOT NULL
              );
          ");
      }

      // ── Gumroad webhook endpoint ────────────────────────────────────────────────
      app.MapPost("/webhook/gumroad", async (HttpContext ctx,
          IHttpClientFactory httpFactory,
          IConfiguration config,
          ILogger<Program> logger) =>
      {
          // Read form data — Gumroad sends application/x-www-form-urlencoded
          var form         = await ctx.Request.ReadFormAsync();
          var sellerId     = form["seller_id"].ToString();
          var orderId      = form["sale_id"].ToString();
          var buyerEmail   = form["email"].ToString();
          var productName  = form["product_name"].ToString();
          var customFields = form["custom_fields[Your GitHub Username]"].ToString().Trim();

          // Validate Gumroad seller ID to reject spoofed requests
          var expectedSellerId = config["Gumroad:SellerId"];
          if (sellerId != expectedSellerId)
          {
              logger.LogWarning("Rejected webhook — invalid seller_id: {Id}", sellerId);
              return Results.Unauthorized();
          }

          if (string.IsNullOrWhiteSpace(customFields))
          {
              logger.LogWarning("Order {OrderId} — no GitHub username provided.", orderId);
              return Results.Ok(new { status = "no_github_username" });
          }

          // Determine tier from product name
          var tier = productName.Contains("Team") ? "Team" :
                     productName.Contains("Updates") ? "Updates" : "Solo";

          // Determine which repos to grant access to
          var repos = tier switch
          {
              "Team"    => new[] { "tensectrakit-core" },
              "Updates" => new[] { "tensectrakit-core" },
              _         => new[] { "tensectrakit-core" }
          };

          // Handle multiple GitHub usernames for team licenses
          var githubUsernames = customFields
              .Split(',', StringSplitOptions.RemoveEmptyEntries | StringSplitOptions.TrimEntries);

          var githubToken = config["GitHub:PersonalAccessToken"];
          var githubOrg   = config["GitHub:Org"] ?? "tensectra-pro";
          var httpClient  = httpFactory.CreateClient();
          httpClient.DefaultRequestHeaders.Add("Authorization", $"Bearer {githubToken}");
          httpClient.DefaultRequestHeaders.Add("User-Agent", "TensectraKit-LicenseServer");
          httpClient.DefaultRequestHeaders.Add("X-GitHub-Api-Version", "2022-11-28");

          foreach (var githubUsername in githubUsernames)
          {
              foreach (var repo in repos)
              {
                  var inviteUrl = $"https://api.github.com/repos/{githubOrg}/{repo}/collaborators/{githubUsername}";
                  var payload   = JsonSerializer.Serialize(new { permission = "pull" });
                  var content   = new StringContent(payload, System.Text.Encoding.UTF8, "application/json");
                  var response  = await httpClient.PutAsync(inviteUrl, content);

                  logger.LogInformation(
                      "GitHub invite: {User} → {Org}/{Repo} — HTTP {Status}",
                      githubUsername, githubOrg, repo, response.StatusCode);
              }

              // Record purchase in database
              var expiresAt = tier == "Updates"
                  ? DateTime.UtcNow.AddDays(365).ToString("O")
                  : (string?)null;

              var licenseKey = GenerateLicenseKey(buyerEmail, tier, config["LicenseServer:MasterSecret"] ?? "change-me");

              using var db = new SqliteConnection($"Data Source={dbPath}");
              await db.ExecuteAsync(@"
                  INSERT INTO Purchases
                      (OrderId, BuyerEmail, GitHubUsername, ProductName, LicenseTier,
                       PurchasedAt, ExpiresAt, LicenseKey)
                  VALUES
                      (@OrderId, @BuyerEmail, @GitHubUsername, @ProductName, @LicenseTier,
                       @PurchasedAt, @ExpiresAt, @LicenseKey)",
                  new {
                      OrderId        = orderId,
                      BuyerEmail     = buyerEmail,
                      GitHubUsername = githubUsername,
                      ProductName    = productName,
                      LicenseTier    = tier,
                      PurchasedAt    = DateTime.UtcNow.ToString("O"),
                      ExpiresAt      = expiresAt,
                      LicenseKey     = licenseKey
                  });
          }

          return Results.Ok(new { status = "success", invited = githubUsernames.Length });
      });

      // ── Expiry check endpoint (called by scheduled job or manual trigger) ──────
      app.MapPost("/admin/revoke-expired", async (
          IHttpClientFactory httpFactory,
          IConfiguration config,
          ILogger<Program> logger) =>
      {
          using var db = new SqliteConnection($"Data Source={dbPath}");
          var expired = (await db.QueryAsync<PurchaseRecord>(@"
              SELECT * FROM Purchases
              WHERE ExpiresAt IS NOT NULL
                AND ExpiresAt < @Now
                AND AccessRevoked = 0",
              new { Now = DateTime.UtcNow.ToString("O") })).ToList();

          var githubToken = config["GitHub:PersonalAccessToken"];
          var githubOrg   = config["GitHub:Org"] ?? "tensectra-pro";
          var httpClient  = httpFactory.CreateClient();
          httpClient.DefaultRequestHeaders.Add("Authorization", $"Bearer {githubToken}");
          httpClient.DefaultRequestHeaders.Add("User-Agent", "TensectraKit-LicenseServer");
          httpClient.DefaultRequestHeaders.Add("X-GitHub-Api-Version", "2022-11-28");

          var revoked = 0;
          foreach (var record in expired)
          {
              var removeUrl = $"https://api.github.com/repos/{githubOrg}/tensectrakit-core/collaborators/{record.GitHubUsername}";
              var response  = await httpClient.DeleteAsync(removeUrl);

              if (response.IsSuccessStatusCode || response.StatusCode == System.Net.HttpStatusCode.NoContent)
              {
                  await db.ExecuteAsync(
                      "UPDATE Purchases SET AccessRevoked = 1 WHERE Id = @Id",
                      new { record.Id });
                  revoked++;
                  logger.LogInformation("Revoked access: {User} (expired {Expiry})",
                      record.GitHubUsername, record.ExpiresAt);
              }
          }

          return Results.Ok(new { checked_count = expired.Count, revoked_count = revoked });
      });

      // ── Health check ────────────────────────────────────────────────────────────
      app.MapGet("/health", () => Results.Ok(new { status = "ok", utc = DateTime.UtcNow }));

      app.Run();

      static string GenerateLicenseKey(string email, string tier, string secret)
      {
          var data = System.Text.Encoding.UTF8.GetBytes($"{email}:{tier}:{DateTime.UtcNow.Year}");
          var key  = System.Text.Encoding.UTF8.GetBytes(secret.PadRight(32)[..32]);
          using var hmac = new System.Security.Cryptography.HMACSHA256(key);
          var hash = hmac.ComputeHash(data);
          return $"TK-{DateTime.UtcNow.Year}-{Convert.ToBase64String(hash)[..32]}";
      }

PART 7 — UPDATES PLAN EXPIRY: STEP-BY-STEP REVOCATION FLOW
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  HOW EXPIRY WORKS IN YOUR SYSTEM
  ─────────────────────────────────
  When a buyer purchases the Updates Plan ($99/year):
    - Webhook fires → ExpiresAt = today + 365 days stored in licenses.db
    - Buyer keeps GitHub collaborator access during this period 

  When ExpiresAt arrives:
    - A daily scheduled call to POST /admin/revoke-expired - fires 
    - The License Server finds all records where ExpiresAt < now AND AccessRevoked = 0
    - It calls GitHub API DELETE to remove the collaborator
    - It sets AccessRevoked = 1 in the database

  HOW TO TRIGGER THE DAILY EXPIRY CHECK
  ────────────────────────────────────────
  You have three options. Use OPTION A for simplicity at launch:

  OPTION A — GitHub Actions Scheduled Workflow  (recommended — free, no extra service)

    Create this file in your tensectrakit-license-server repo:
    Path: .github/workflows/revoke-expired.yml

        name: Revoke Expired Licenses
        on:
          schedule:
            - cron: '0 2 * * *'   # runs daily at 02:00 UTC
          workflow_dispatch:       # allows manual trigger from GitHub Actions UI

        jobs:
          revoke:
            runs-on: ubuntu-latest
            steps:
              - name: Call revoke endpoint
                run: |
                  curl -X POST \
                    -H "Authorization: Bearer ${{ secrets.ADMIN_TOKEN }}" \
                    https://tensectrakit-license-server.up.railway.app/admin/revoke-expired

    In your GitHub repo → Settings → Secrets → Actions:
      Add secret: ADMIN_TOKEN = any long random string

    In your License Server Program.cs, protect the endpoint:
        app.MapPost("/admin/revoke-expired", async (...) =>
        {
            var authHeader = ctx.Request.Headers["Authorization"].ToString();
            var adminToken = config["AdminToken"] ?? "";
            if (authHeader != $"Bearer {adminToken}")
                return Results.Unauthorized();
            // ... rest of revoke logic
        });

    Add to Railway env vars:
        AdminToken = same_long_random_string_as_above

  OPTION B — Railway Cron Job  (paid feature on Railway Pro)

    Railway Pro allows scheduled "Cron" services that call HTTP endpoints on a schedule.
    Set it to call POST /admin/revoke-expired daily at 02:00 UTC.

  OPTION C — Manual (Phase 1, before you have many Updates Plan subscribers)

    While you have fewer than 20 Updates Plan subscribers:
    - Keep a spreadsheet: GitHub Username | ExpiresAt
    - Set a monthly calendar reminder to check for expired subscriptions
    - Manually remove collaborators in GitHub:
        GitHub → tensectra-pro org → tensectrakit-core repo
        → Settings → Collaborators → find the username → Remove

  STEP-BY-STEP: WHAT TO DO WHEN UPDATES PLAN EXPIRES
  ─────────────────────────────────────────────────────────────────────────────────────────────

  Day -14 (2 weeks before expiry):
    □ The License Server's daily check finds ExpiresAt is within 14 days
    □ Send a renewal reminder email manually (or automate via SendGrid):
      Subject: "Your TensectraKit Updates Plan renews in 14 days"
      Body:    "Click here to renew for another year: [Gumroad link]
                Your access remains active until [ExpiresAt date].
                After that date, you will still have your generated code,
                but will not receive new template updates."

  Day 0 (ExpiresAt date):
    □ GitHub Actions cron job runs POST /admin/revoke-expired
    □ License Server removes GitHub collaborator access for expired buyers
    □ AccessRevoked = 1 is set in the database
    □ Buyer can no longer git pull from tensectrakit-core
    □ Their EXISTING generated projects are NOT affected — they own that code

  Day +1 (if buyer renews):
    □ Buyer purchases Updates Plan again on Gumroad
    □ Gumroad fires webhook
    □ License Server sets new ExpiresAt = today + 365 days
    □ GitHub API re-invites the collaborator
    □ Buyer accepts invitation (or it auto-accepts if they were previously a member)
